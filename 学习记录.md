function Point(x,y){
    this.x=x;
    this.y=y;
}
Point.prototype.move=function{
    this.x+=x;
    this.y+=y;
} //在Point构造函数的原型对象上创建一个move方法

var p=new Point(0,0)//实例化的p函数可以访问move方法
